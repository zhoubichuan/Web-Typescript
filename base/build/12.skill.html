# 12.泛型

/* 泛型：可以支持不特定的数据类型。要求：传入的参数和返回的参数一致 */
function _getData<T>(value: T): T {
  return value;
}
_getData<number>(123);
_getData<string>("1212");

function _getData2<T>(value: T): any {
  return "asdfasd";
}
_getData2<number>(124);
_getData2<string>("asdfasd");

//泛型类：
class MinClass {
  public list: number[] = [];
  add(num: number) {
    this.list.push(num);
  }
  min(): number {
    var minNum = this.list[0];
    for (var i = 0; i < this.list.length; i++) {
      if (minNum > this.list[i]) {
        minNum = this.list[i];
      }
    }
    return minNum;
  }
}

var m = new MinClass();
m.add(3);
m.add(22);
alert(m.min());

//类的泛型
class MinClas<T> {
  public list: T[] = [];
  add(value: T): void {
    this.list.push(value);
  }
  min(): T {
    var minNum = this.list[0];
    for (var i = 0; i < this.list.length; i++) {
      if (minNum > this.list[i]) {
        minNum = this.list[i];
      }
    }
    return minNum;
  }
}
var m1 = new MinClas<number>();
m1.add(12);
m1.add(3);
alert(m1.min());
var m2 = new MinClas<string>();
m2.add("c");
m2.add("a");
m2.add("v");
alert(m2.min());
